  for (i in 1:N) {
    for (j in 1:ninter) {
      outcome[i, j] ~ dcat(prob)
    }
  }
  prob[1] <- 1.0 / ((1 + exp(theta)) * (1 + exp(eta)))
  prob[2] <- exp(theta) / ((1 + exp(theta)) * (1 + exp(eta)))
  prob[3] <- exp(eta) / (1 + exp(eta))
  theta ~ dnorm(-0.5, 0.00001)T(cutoff, )
  eta ~ dnorm(-2, 0.00001)T(, cutoff2)
}
