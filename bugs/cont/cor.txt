# 3 outcomes clustering groups with continuous activity

model { 
  for (i in 1:N) {
    for (j in 1:ninter) {
      response[i, j] ~ dbern(1 - prob[i, j])
      prob[i, j] <- pnorm(0, mu1[i] + rho * (activity[i, j] - mu2[i]) * sqrt(prec), 1.0 / (1 - rho * rho))
      activity[i, j] ~ dnorm(mu2[i], prec)
    }
    mu1[i] ~ dnorm(0, 0.00001)
    mu2[i] ~ dnorm(0, 0.00001)
  }
  rho ~ dunif(-1, 1)
  prec ~ dgamma(0.001, 0.001)
}
