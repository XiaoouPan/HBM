# 3 outcomes clustering groups with continuous activity

model { 
  for (j in 1:ninter) {
    response[j] ~ dbern(1 - prob[j])
    prob[j] <- pnorm(0, mu1 + rho * (activity[j] - mu2), 1.0 / (1 - rho * rho))
    activity[j] ~ dnorm(mu2, 1)
  }
  rho ~ dunif(0, 1)
  mu1 ~ dnorm(-2, 0.00001)T(, cutoff)
  mu2 ~ dnorm(1, 0.00001)T(, cutoff2)
}
